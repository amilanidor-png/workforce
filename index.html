      function taskSheet(id){
      const t=S.tasks.find(x=>x.id===id); if(!t) return;
      const u=user(t.assignee);
      const canEdit = isMgr() || t.assignee===S.me;
      openSheet(`
      <h2>${t.title}</h2>
      <p style="color:var(--muted);margin:2px 0 14px">${fmtDate(t.date)} · ${t.time}</p>
      <span class="person-tag"><span class="pa">${initials(u.name)}</span>${u.name}</span>
      <span class="badge ${t.priority==='גבוהה'?'b-red':'b-blue'}">${t.priority}</span>
      ${t.notes?`<div class="card" style="margin-top:14px;background:var(--surface)"><div style="font-size:13px;color:var(--muted);margin-bottom:4px">הערות</div>${t.notes}</div>`:''}
      ${canEdit?`
      <label class="fl">עדכון סטטוס</label>
      <div class="seg">
      ${['ממתינה','בתהליך','הושלמה'].map(s=>`<label><input type="radio" name="ts" value="${s}" ${t.status===s?'checked':''}><span class="opt">${s}</span></label>`).join('')}
      </div>
      <button class="btn btn-in" style="margin-top:16px" id="t_save"><i class="ti ti-check"></i> שמור</button>`:''}
      ${isMgr()?`<button class="btn btn-ghost" style="margin-top:8px;color:var(--red-600);border-color:var(--red-400)" id="t_del"><i class="ti ti-trash"></i> מחק משימה</button>`:''}
      <button class="btn btn-ghost" style="margin-top:8px" onclick="closeSheet()">סגור</button>
      `);
      if(canEdit && $('#t_save')) $('#t_save').onclick=()=>{
      t.status=document.querySelector('input[name=ts]:checked').value; save(); closeSheet(); render(); toast('הסטטוס עודכן');
      };
      if($('#t_del')) $('#t_del').onclick=()=>{
      S.tasks=S.tasks.filter(x=>x.id!==id); save(); closeSheet(); render(); toast('המשימה נמחקה');
      };
      }
      /* ---------- Live clock ---------- */
      setInterval(()=>{ const c=$('#liveClock'); if(c) c.textContent=clockNow(); },1000);

      renderHeader(); renderTabs(); render();
      window.closeSheet=closeSheet;
